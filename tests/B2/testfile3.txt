void move(int x, int y)
{
	printf("%d->%d\n", x, y);
	return ;
}
void hanoi(int n, int a , int b , int c)
{
	if (n == 1)
	{
		move(a, c);
	}
	else
	{
		hanoi(n - 1, a, c, b);
		move(a, c);
		hanoi(n - 1, b, a, c);
	}
	return ;
}
