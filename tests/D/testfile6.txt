int n;
const int maxn = 1005;
int x[1005];
int a[1005], b[1005], c[1005];
int cnt[1005 * 4];

void sortA(int a[]) {
    int i, j;
    for (i = 0; i < n; i = i + 1)
        for (j = i + 1; j < n; j = j + 1)
            if (a[i] > a[j]) {
                int t = a[i];
                a[i] = a[j];
                a[j] = t;
            }
}

void sortB(int a[]) {
    int i, mx = -100;
    for (i = 0; i < n; i = i + 1) {
        cnt[a[i]] = cnt[a[i]] + 1;
        if (a[i] > mx) mx = a[i];
    }
    int j, now = 0;
    for (i = 0; i <= mx; i = i + 1) {
        for (j = cnt[i]; j; j = j - 1) {
            a[now] = i;
            now = now + 1;
        }
    }
}

void sortC(int a[]) {
    int i;
    for (i = 0; i < n; i = i + 1) {
        int id = i, j;
        for (j = i + 1; j < n; j = j + 1)
            if (a[j] < a[id])
                id = j;
        int t = a[i];
        a[i] = a[id];
        a[id] = t;
    }
}

int main() {
    n = getint();
    int i;
    for (i = 0; i != n; i = i + 1) {
        a[i] = getint();
        b[i] = a[i];
        c[i] = b[i];
    }
    sortA(a);
    sortB(b);
    sortC(c);
    for (i = 0; n - i; i = i + 1) {
        b[i] = b[i] - a[i];
        c[i] = c[i] - b[i] - a[i];
    }
    for (i = 0; i - n; i = i + 1) {
        if (b[i]) {
            printf("Bad b[%d]!\n", i);
            return 0;
        }
        if (c[i]) {
            printf("Bad c[%d]!\n", i);
            return 0;
        }
    }
    printf("Gracefully terminated!\n");
    return 0;
}
